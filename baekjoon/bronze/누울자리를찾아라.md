function solution(count, input) {
  let row = 0, column = 0; 

  for (let i=0; i<count; i++) {
    if (input[i].includes('..')) row++; 
    
    for (let j=0; j<count; j++) {
      if (i === 0) continue;
      if (input[i-1][j] === '.' && input[i][j] === '.') {
        column++;
        break;
      }
    }
  }

  return `${row} ${column}`
}
