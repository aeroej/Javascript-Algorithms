const fs = require('fs');

  ? fs.readFileSync('/dev/stdin').toString()
  : `4 2
1 3 1 1`
).split('\n');

let [N, M] = stdin[0].split(' ').map(Number);

let result = 0;
let left = 0, right = 1;
let sum = input[left] + input[right];

while(left < N && right < N) {
  if (sum === M) {
    result++;
    sum += input[++right];
  }
  else if (sum < M) sum += input[++right];
  else if (sum > M) sum -= input[left++];
}

console.log(result);
