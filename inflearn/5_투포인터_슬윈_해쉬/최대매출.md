```javascript
function solution(arr, k) {
  let left = 0, right = k-1;
  let sum = 0;

  for (let i=0; i<k; i++) sum += arr[i];
  let max = sum;

  while (right < arr.length) {
    sum -= arr[left++];
    sum += arr[++right];
    max = sum > max ? sum : max;
  }

  return max;
}
```
