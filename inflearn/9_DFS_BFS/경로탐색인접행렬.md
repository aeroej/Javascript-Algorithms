```javascript
let n = 5;
let arr = [[1, 2], [1, 3], [1, 4], [2, 1], [2, 3], [2, 5], [3, 4], [4, 2], [4, 5]];
let visited = Array.from({ length: n + 1 }, () => false);

let res = [];
let cnt = 0;

function DFS(v) {
  visited[v] = true;
  res.push(v);

  if (v === n) {
    cnt++;
    return;
  }

  for (let x of arr) {
    if (v === x[0] && !visited[x[1]]) {
      DFS(x[1]);
      res.pop();
      visited[x[1]] = false;
    }
  }
}

DFS(1);
console.log(cnt);
```
