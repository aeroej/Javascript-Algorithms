```javascript
let score = [10, 25, 15, 6, 7];
let time = [5, 12, 8, 3, 4];
let limitTime = 20;
let totalScore = 0, totalTime = 0;
let max = Number.MIN_SAFE_INTEGER;

function DFS(n) {
  if (n >= score.length) {
    if (totalTime <= limitTime) max = Math.max(max, totalScore);
    return;
  }

  totalScore += score[n];
  totalTime += time[n];
  DFS(n + 1);

  totalScore -= score[n];
  totalTime -= time[n];
  DFS(n + 1);
}

DFS(0);
console.log(max);
```
