```javascript
function solution(clothes) {
  let sum = 1;
  
  let hash = new Map();
  clothes.forEach(cloth => {
    hash.set(cloth[1], (hash.get(cloth[1]) | 0) + 1);
  })

  for (let [key, val] of hash) {
    sum *= val + 1;
  }

  return sum - 1;
}

console.log(solution([["yellowhat", "headgear"], ["bluesunglasses", "eyewear"], ["green_turban", "headgear"]]))
```
